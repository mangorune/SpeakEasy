% left indent, right indent, text
\newcommand{\circTopText}[3]{%
  [postaction={%
    decorate,
    decoration={%
      text format delimiters={|}{|},
      text along path, text align/align=center,
      % text align/fit to path stretching spaces=true,
      raise=-0.5\baselineskip, reverse path=true,
      text align/left indent={#1},
      text align/right indent={#2},
      text={#3}
    }
  }]
}
\newcommand{\circBottomText}[3]{%
  [postaction={%
    decorate,
    decoration={%
      text format delimiters={|}{|},
      text along path, text align/align=center,
      % text align/fit to path stretching spaces=true,
      raise=0\baselineskip, reverse path=false,
      text align/left indent={#1},
      text align/right indent={#2},
      text={#3}
    }
  }]
}
